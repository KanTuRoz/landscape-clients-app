// Renderização com distância (CORRIGIDA) function render() { const searchTerm = document.getElementById('search').value.toLowerCase(); const listDiv = document.getElementById('list'); listDiv.innerHTML = ''; if (!showList) return; const clients = getClients(); if (clients.length === 0) { listDiv.innerHTML = '

No clients yet.

'; return; } // Se não houver permissão de localização, apenas mostra os cards sem distância if (!navigator.geolocation) { // Itera sobre o array original, mas só mostra os que passam no filtro clients.forEach((c, originalIndex) => { if (c.name.toLowerCase().includes(searchTerm) || c.eircode.toLowerCase().includes(searchTerm)) { const card = createClientCard(c, originalIndex, null); listDiv.appendChild(card); } }); return; } navigator.geolocation.getCurrentPosition( pos => { const { latitude, longitude } = pos.coords; // Itera sobre o array original, filtrando clients.forEach((c, originalIndex) => { if (c.name.toLowerCase().includes(searchTerm) || c.eircode.toLowerCase().includes(searchTerm)) { const card = createClientCard(c, originalIndex, { lat: latitude, lon: longitude }); listDiv.appendChild(card); // Buscar distância via OSRM usando o eircode fetch(`https://router.project-osrm.org/route/v1/driving/${longitude},${latitude};${c.eircode}?overview=false`) .then(r => r.json()) .then(data => { if (data.routes && data.routes[0]) { const km = (data.routes[0].distance / 1000).toFixed(1); const distSpan = document.getElementById(`dist-${originalIndex}`); if (distSpan) distSpan.innerText = `📏 ${km} km`; } }) .catch(() => {}); } }); }, err => { // Em caso de erro de localização, mostra sem distância clients.forEach((c, originalIndex) => { if (c.name.toLowerCase().includes(searchTerm) || c.eircode.toLowerCase().includes(searchTerm)) { const card = createClientCard(c, originalIndex, null); listDiv.appendChild(card); } }); } ); } // Função createClientCard (ajustada para receber o índice original) function createClientCard(client, originalIndex, coords) { const div = document.createElement('div'); div.className = 'client'; div.innerHTML = ` ${client.name}
${client.eircode}
${client.notes ? `📝 ${client.notes}
` : ''}
${coords ? '📏 calculando...' : ''}
`; return div; }
